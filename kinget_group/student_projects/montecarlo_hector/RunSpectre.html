# MCRunSpectre      - Monte Carlo Simulation Runner
#                     runs spectre on all netlists in the results directory

# usage:       'MCRunSpectre.pl'

#constants:
my $RESULTPATH = "results/";      <-Looks into this result path for .scs files

#get list of all netlists in results directory
$files = `ls $RESULTPATH*.scs | xargs`;      <-- Uses 'xargs' from the local shell  to match and pipe all .scs files

#for each netlist, run spectre and output a file in ascii format
foreach $file (split(' ',$files)) {
    print "$file\n";
    $file =~ m/(.*)\.scs/;
   
    print `spectre -format nutascii -raw $1.out $file`;         <--Run spectre and output them into a .out file

}
